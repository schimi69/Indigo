PROJECT(Graph)

add_library(graph OBJECT ${Graph_src} ${Graph_headers})

if(UNIX OR APPLE)

endif()

set_target_properties(graph PROPERTIES COMPILE_FLAGS "${COMPILE_FLAGS}")
